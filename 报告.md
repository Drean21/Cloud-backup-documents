### （三）从 VisionSeed 获取数据

**step：**

1. 创建`example0.py`文件
2. 引入 `visionseed` 库文件和 `serial` 串口库
3. 实例化一个` YtDataLink`
4. 编写函数不断的调用 `YtDataLink` 的 `recvRunOnce` 方法从 `VisionSeed` 获取消息
5. 获取更多数据：增加人脸姿态信息的打印代码、 增加人脸五官配准点的打印
6. 使用vscode运行`example0.py`程序

阶段代码：

~~~python
# 引入 VisionSeed 库
from visionseed import YtVisionSeed, YtDataLink
# 引入串口库
import serial
# 实例化一个 YtDataLink
vs = YtVisionSeed(serial.Serial("COM3", 115200, timeout=0.5))


# 编写函数不断的调用 YtDataLink 的 recvRunOnce 方法从 VisionSeed 获取消息
def main():
    # 一直执行
    while True:
        # 从VisionSeed获取消息
        result, msg = vs.recvRunOnce()
        # 如果存在结果
        if result:
            # 实例化对象并调用API取出face_detection，用count接收
            YtVisionSeedModel = YtDataLink.YtVisionSeedModel
            count = result.getResult([YtVisionSeedModel.FACE_DETECTION])
            # 遍历从count取出信息数据
            for i in range(count):
                # 定义字符串变量line用来从count中按格式取出具体数据
                line = ''
                # 获取检测框
                rect = result.getResult([YtVisionSeedModel.FACE_DETECTION, i])
                # 如果检测框存在，按指定格式拼接赋给字符串变量line
                if (rect):
                    line += 'face detected rect: (%d, %d, %d, %d) ' % (
                        rect.x, rect.y, rect.w, rect.h)
                    # 打印脸部姿态信息
                    pose = result.getResult(
                        [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_POSE])
                    if not (pose is None):
                        line += 'pose(roll,yaw,pitch): (%5.1f, %5.1f, %5.1f) ' % (
                            pose.array[0], pose.array[1], pose.array[2])
                    # 增加人脸五官配准点的打印,获取 90 点关键点
                    shape = result.getResult(
                        [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_LANDMARK])
                    if (shape):
                        faceShape = shape.faceShape
                        line += 'left pupil: x:(%d), y:(%d)' % (
                            faceShape.pupil[0].x, faceShape.pupil[0].y)
                # 如果检测框不存在，打印输出'no face detected'
                else:
                    line = 'no face detected'
                # 每遍历count中一个元素则取出打印一次
                print(line)


# 最后在main中调用recvMsg函数
if __name__ == "__main__":
    main()
~~~

- **使用眼部配准点信息和姿态信息完成一个疲劳监测程序**

  代码如下：

  ```python
  # 引入库文件，实例化 vs
  from visionseed import YtVisionSeed, YtDataLink
  import serial
  import time
  vs = YtVisionSeed(serial.Serial("COM3", 115200, timeout=0.5))
  
  
  # 使用眼睛睁开程度判断是否疲劳，定义阈值threshold = 0.1
  threshold = 0.1
  # 通过眼睛的睁开程度判断是否疲劳的方法
  def isExhaustedSimple(result):
      # 获取 90 点关键点
      YtVisionSeedModel = YtDataLink.YtVisionSeedModel
      shape = result.getResult([YtVisionSeedModel.FACE_DETECTION, 0, YtVisionSeedModel.FACE_LANDMARK])
      # 如果检测到脸型区域执行下列语句
      if (shape):
          faceShape = shape.faceShape
          # 根据眼睛上下点的距离与左右眼角的距离比值得到眼睛睁开的程度
          leftEyeOpenness = (faceShape.leftEye[6] - faceShape.leftEye[2]).length() / (
              faceShape.leftEye[0] - faceShape.leftEye[4]).length()
          rightEyeOpenness = (faceShape.rightEye[6] - faceShape.rightEye[2]).length() / (
              faceShape.rightEye[0] - faceShape.rightEye[4]).length()
          # 眼睛的睁开程度
          eyeOpenness = (leftEyeOpenness + rightEyeOpenness) / 2
          # 判断眼睛是否闭着的阈值，眼睛的睁开程度小于该数字就判定为紧闭
          return eyeOpenness < threshold
      # 如果检测不到脸部区域直接返回False    
      return False
  
  
  # 不断的从 VisionSeed 接收数据并判断打印疲劳状态
  def recvMsg():
      # 一直执行
      while True:
          # 从VisionSeed获取消息
          result, msg = vs.recvRunOnce()
          # 如果存在结果
          if result:
              # 实例化对象并调用API取出face_detection，用count接收
              YtVisionSeedModel = YtDataLink.YtVisionSeedModel
              count = result.getResult([YtVisionSeedModel.FACE_DETECTION])
              if count > 0:
                  # 判断是否疲劳
                  exhausted = isExhaustedSimple(result)
                  # 如果疲劳，则打印疲劳状态
                  if exhausted:
                      print('you are exhausted')
  
  
  if __name__ == "__main__":
      recvMsg()
  ```



- **用RingBuffer优化疲劳监测程序**

  代码如下：

  ```python
  # 引入库文件（包含优化监测程序用到的numpy和numpy_ringbuffer这两个库），实例化 vs
  from visionseed import YtVisionSeed, YtDataLink
  import serial
  import time
  import numpy as np
  from numpy_ringbuffer import RingBuffer
  vs = YtVisionSeed( serial.Serial("COM3",115200,timeout=0.5) )
  eyeClosedRingBuffer = RingBuffer(capacity=5, dtype=np.bool)
  # 使用眼睛睁开程度判断是否疲劳
  threshold = 0.1
  # 定义isExhausted，使用RingBuffer优化程序
  def isExhausted(result):
      def eyeReallyClosed():
          # ringbuffer 中的所有记录都为闭眼才判定为真的闭眼，以避免对眨眼的误判
          for close in eyeClosedRingBuffer:
              if not close:
                  return False
          return True
      # 获取 90 点关键点
      YtVisionSeedModel = YtDataLink.YtVisionSeedModel
      shape = result.getResult([YtVisionSeedModel.FACE_DETECTION, 0, YtVisionSeedModel.FACE_LANDMARK])
      if (shape):
          faceShape = shape.faceShape
          #line += 'left pupil: x:(%d), y:(%d)' % (faceShape.pupil[0].x, faceShape.pupil[0].y)
          # 根据眼睛上下点的距离与左右眼角的距离比值得到眼睛睁开的程度
          leftEyeOpenness = (faceShape.leftEye[6] - faceShape.leftEye[2]).length() / (faceShape.leftEye[0] - faceShape.leftEye[4]).length()
          rightEyeOpenness = (faceShape.rightEye[6] - faceShape.rightEye[2]).length() / (faceShape.rightEye[0] - faceShape.rightEye[4]).length()
          pose = result.getResult([YtVisionSeedModel.FACE_DETECTION, 0, YtVisionSeedModel.FACE_POSE])
          if not (pose is None):
              yawAngle = pose.array[1]
              # 根据脸部的左右偏转角度决定使用左眼或者右眼的睁开程度（选择离镜头最近的眼睛的睁开程度）
              eyeOpenness = leftEyeOpenness if yawAngle < 0 else rightEyeOpenness
          else:
              eyeOpenness = (leftEyeOpenness + rightEyeOpenness) / 2
          # 判断眼睛是否闭着的阈值，眼睛的睁开程度小于该数字就判定为紧闭
          eyeClosedRingBuffer.append(eyeOpenness < threshold)
          # 判断是否真的紧闭
          return eyeReallyClosed()
      return False
  
  
  
  # 不断的从 VisionSeed 接收数据并判断打印疲劳状态
  def recvMsg():
      while True:
          result, msg = vs.recvRunOnce()
          if result:
              YtVisionSeedModel = YtDataLink.YtVisionSeedModel
              count = result.getResult([YtVisionSeedModel.FACE_DETECTION])
              if count>0:
                  # 判断是否疲劳
                  exhausted = isExhausted(result)
                  # 如果疲劳，则打印疲劳状态
                  if exhausted:
                      print('you are exhausted')
  if __name__ == "__main__":
      recvMsg()
  
  ```

#### 2、编写`create_database.py`程序创建数据库表

**代码如下：**

~~~python
import mysql.connector # 导入数据库连接依赖
from mysql.connector import Error

# 定义连接数据库的参数
database = 'exhausted_state_monitor'
tableName = 'exhausted_state'
# 连接本地主机创建数据库表
def createDb(host='localhost', user='root', password='root'): 
    # 捕获可能出现的数据库异常
    try:
        # 获取数据库连接对象
        connection = mysql.connector.connect(host=host,
                                            user=user,
                                            password=password)
        # 成功获取数据库连接对象后
        if connection.is_connected():
            # 获取数据库SQL执行对象
            cursor = connection.cursor()
            # 定义SQL语句
            sql = """CREATE DATABASE IF NOT EXISTS {};
                    USE {};
                    CREATE TABLE IF NOT EXISTS {} (
                        time DATETIME PRIMARY KEY,
                        name VARCHAR(255),
                        face_rect_x SMALLINT,
                        face_rect_y SMALLINT,
                        face_rect_h SMALLINT,
                        face_rect_w SMALLINT
                    );""".format(database, database, tableName)
            # 执行sql语句并打印输出执行效果        
            for result in cursor.execute(sql, multi=True):
                if result.with_rows:
                    print("Rows produced by statement '{}':".format(result.statement))
                    print(result.fetchall())
                else:
                    print("Number of rows affected by statement '{}': {}".format(
                    result.statement, result.rowcount))
            cursor.close()
    # 将捕获到的异常打印输出       
    except Error as e:
        print('error while create database', e)
    # 无论如何都要执行下列语句关闭数据库连接    
    finally:
        if connection.is_connected():
            connection.close()

if __name__ == "__main__":
    createDb()
~~~

#### 3、在`get_visionseed_data.py`的基础上添加`logto`方法将获得数据写入到数据库

**代码如下：**

~~~python
"""文档说明：
    利用眼部配准点计算人眼的开闭状态结合姿态及使用RingBuffer缓存优化实现基础的疲劳监测功能,并将所得数据保存到本地数据库
"""
from visionseed import YtVisionSeed, YtDataLink  # 调用sdk依赖包
import serial  # 串口库
import datetime  # 获取当前时间数据的依赖包
import numpy as np  # 矩阵计算的函数库
from numpy_ringbuffer import RingBuffer  # 增加环形buffer进行缓存一段时间的状态
import pymysql  # 数据库连接依赖库
import re  # 字符串正则匹配依赖库
import cv2 as cv  # 利用opencv进行视频图像输出

# 实例化YTdataLink类
datalink = YtVisionSeed(serial.Serial("COM3", 115200, timeout=0.5))

"""定义一个capacity为5的RingBuffer，用来存储一段时间疲劳状态,用于判断,不存储的话一闭眼就被误报检测为劳累
   适当调节RingBuffer的capacity以在延迟和误判之间取得平衡。
   RingBuffer相当于对数据进行了滤波处理，这样使得程序的响应变慢了。可以考虑调整RingBuffer的长度在实时性和避免误判之间平衡。
"""
eyeClosedRingBuffer = RingBuffer(capacity=5, dtype=np.bool)

"""
只要检测到疲劳就立即会写入数据库，当人眼紧闭的时候则会持续不断的写入数据库，所以我们对写入的频率进行限制
"""
leastTimeToRecordState = 1.5  # second为单位

# 读取摄像头数据（具体参数以自身电脑测试为准）
cap = cv.VideoCapture(0)  # 0是读取本地电脑自带摄像头的数据,改为1读取visionseed数据
if (cap.isOpened() == False):
    print("不能读取摄像头数据")
    exit(0)

"""数据保存至本地数据库"""


def logto(strnow, name, x, y, w, h):
    # 使用pymysql.connect方法连接数据库
    db = pymysql.connect(host='127.0.0.1', port=3306, user='root', password='root',
                     database='exhausted_state_monitor', charset='utf8')
    # 操作数据库，获取db下的cursor对象
    cursor = db.cursor()
    # 使用cursor.execute来执行SQL语句
    # 提交表单内条写入数据库
    sql = "insert into exhausted_state (time,name,face_rect_x,face_rect_y,face_rect_h,face_rect_w)values(%s,%s,%s,%s,%s,%s)"
    try:
        # 执行sql语句
        cursor.execute(sql, (strnow, name, x, y, w, h))
        # 提交到数据库执行
        print("已保存一条劳累的记录")
        db.commit()
        # print("已保存一条劳累的记录")
    except Exception as err:
        # 检查异常原因是否是感兴趣的
        result1 = re.search('Duplicate entry.*key.*PRIMARY', str(err))
        # 如果是，什么都不用做
        # 否则（也不知道是什么原因），那就回滚吧
        if (result1 == None):
            # 如果发生错误则回滚
            db.rollback()
    # 关闭数据库连接
    db.close()


def main():
    # 增加一个RingBuffer缓存进行优化的函数
    def eyeReallyClosed():
        # 当ringbuffer中的所有记录都为闭眼时才判定为真的闭眼，以避免对眨眼的误判
        for close in eyeClosedRingBuffer:
            # 利用for循环遍历,如果存在是not true成立,即有存在不是闭眼的情况,就是返回False,不是真的劳累
            if not close:
                return False
        return True  # 否则就是buffer缓存中都是True，即都是闭眼,返回True,是真的劳累

    # 初始化上一次写入数据库的时间戳
    lastLogTime = datetime.datetime.now()
    while True:
        # 显示摄像头数据
        if (cap.isOpened()):
            ret, frame = cap.read()
            cv.namedWindow('frame', cv.WINDOW_NORMAL)
            cv.imshow('frame', frame)
            k = cv.waitKey(1)
            if k == 27:
                cap.release()
                cv.destroyAllWindows()
                break

        result, msg = datalink.recvRunOnce()
        if result:
            YtVisionSeedModel = YtDataLink.YtVisionSeedModel
            count = result.getResult([YtVisionSeedModel.FACE_DETECTION])
            for i in range(count):
                line = ''
                # 获取人脸识别结果
                faceName = result.getResult(
                    [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_RECOGNITION])
                if (faceName):
                    line += 'name: %s (confidence: %.3f) ' % (faceName.str,
                                                              faceName.conf)
                shape = result.getResult(
                    [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_LANDMARK])
                if (shape):
                    # 计算眼睛的睁开程度,根据眼睛上下点的距离与左右眼角的距离比值得到眼睛睁开的程度
                    faceShape = shape.faceShape
                    leftEyeOpenness = (faceShape.leftEye[6] - faceShape.leftEye[2]).length() / (
                        faceShape.leftEye[0] - faceShape.leftEye[4]).length()
                    rightEyeOpenness = (faceShape.rightEye[6] - faceShape.rightEye[2]).length() / (
                        faceShape.rightEye[0] - faceShape.rightEye[4]).length()
                    # 增加使用头部姿态信息yaw角度进行优化,比如当人左倾,使用右眼的信息判断疲劳,当人右倾,使用左眼的信息判断疲劳
                    pose = result.getResult(
                        [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_POSE])
                    if not(pose is None):
                        yawAngle = pose.array[1]
                        print('头部左偏,正通过您的右眼进行判断' if yawAngle <
                              0 else '头部右偏,正通过您的左眼进行判断', yawAngle)
                        eyeOpenness = leftEyeOpenness if yawAngle < 0 else rightEyeOpenness
                    else:
                        eyeOpenness = (leftEyeOpenness + rightEyeOpenness) / 2
                    # 设置判断眼睛是否闭着的‘阈值’，当眼睛的睁开程度小于该数字就判定为紧闭
                    threshold = 0.1
                    # append方法 增加比较结果的bool值存入Buffer缓存中.
                    # 这里的计算结果小于阈值是True成立的话即是闭眼,false则不是闭眼
                    eyeClosedRingBuffer.append(eyeOpenness < threshold)
                    """前面基础上新增加的代码如下"""
                    # 获取姓名
                    name = result.getResult(
                        [YtVisionSeedModel.FACE_DETECTION, i, YtVisionSeedModel.FACE_RECOGNITION])
                    if (name):
                        print("识别人的姓名是:" + name.str)
                        name = name.str
                    else:
                        print("识别的人未知")
                    # 获取人脸坐标
                    rect = result.getResult(
                        [YtVisionSeedModel.FACE_DETECTION, i])
                    if (rect):
                        x = rect.x
                        y = rect.y
                        w = rect.w
                        h = rect.h
                        print("检测到人脸,x坐标{},y坐标,宽度{},高度{}".format(x, y, w, h))
                    # 判断是否真的劳累状态下的闭眼的布尔值
                    if eyeReallyClosed():
                        print("你是劳累的,将会把记录写入数据库")
                        # 获取当前的时间
                        now = datetime.datetime.now()
                        print(now)
                        # 将时间格式转化为数据库支持的datetime格式
                        strnow = now.strftime('%Y-%m-%d %H:%M:%S')
                        # 只在给定的时间间隔之后才将疲劳状态写入数据库
                        if (now - lastLogTime).seconds >= leastTimeToRecordState:
                            logto(strnow, name, x, y, w, h)
                    else:
                        print("你并不劳累")


if __name__ == "__main__":
~~~

~~~python
import flask  # python的web开发依赖库

import pymysql  # 提取mysql数据


class Database:
    # 获取数据库连接对象
    def __init__(self):
        host = '127.0.0.1'  # 本地数据库地址
        port = 3306
        user = 'root'
        password = 'root'
        database = 'exhausted_state_monitor'
        charset = 'utf8'  # 设置编码
        self.con = pymysql.connect(host=host, port=port, user=user, password=password,
                                   database=database, cursorclass=pymysql.cursors.DictCursor, charset=charset)

    # 展示所有人不包括特定用户的疲劳信息
    def showData_all(self):
        tablename = "exhausted_state"
        try:
            # with关键字后面才会自动关闭cursor()对象
            with self.con.cursor() as cursor:
                sql = "select * from {}".format(tablename)
                print("执行的sql语句为:" + sql)
                cursor.execute(sql)
                result = cursor.fetchall()
                return result
        finally:
            # 关闭数据库连接
            self.con.close()

    # 用户过滤,只展示特定用户的疲劳信息
    def showData(self, name):
        tablename = "exhausted_state"
        try:
            with self.con.cursor() as cursor:
                # 读取数据库中名字为name的所有记录，增加where关键字进行限制
                sql = "select * from {} where name = '{}'".format(
                    tablename, name)
                print("执行的sql语句为:"+sql)
                cursor.execute(sql)
                result = cursor.fetchall()
                return result
        finally:
            self.con.close()


# 初始化
app = flask.Flask(__name__)


# 设置虚拟路径和数据请求方式
@app.route('/', methods=['GET', 'POST'])
def index():
    # 定义查找全部用户信息的SQL逻辑
    def db_query():
        db = Database()
        result = db.showData_all()
        return result
    res = db_query()
    # 定义指定用户查找的SQL逻辑

    def db_query_name(x):
        db = Database()
        result = db.showData(x)
        return result
    # 用户过滤：默认显示全部用户的疲劳状态信息。搜索时如果对应用户在疲劳监测表中有记录，返回其记录，若无则告知"没有此人"
    x = flask.request.values.get('value')
    info = ''
    print(x)
    if x:
        if db_query_name(x):
            res = db_query_name(x)
        else:
            info = '没有此人'
    else:
        res = db_query()
    return flask.render_template('my_web.html', result=res, info=info, content_type='application/json')


# 运行
app.debug = True
app.run()
~~~
